  4. Even Odd

    CODE :

                    const findEvenOdd = (N, Arr) => 
                {
                  let even = 0;
                  for(i=0;i< N;i++){
                    if(Arr[i]%2==0){
                      even=even+Arr[i];
                    }
                  }
                  let odd =0;
                  for(j=0;j< N;j++){
                    if(Arr[j]%2!==0){
                      odd=odd+Arr[j];
                    }
                  }
                  let arr=[even,odd];
                  return arr;
                };

    OUTPUT :

                    Input
                7
                1 2 3 4 5 6 7
                
                Output
                12 16
